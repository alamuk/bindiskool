          </svg>
        `,
        caption: "Helped others 2–3× revenue"
      }
    ];

    function logoSrcFor(item) {
      if (item.domain) {
        return `https://logo.clearbit.com/${item.domain}`;
      }
      return null;
    }

    function renderAwards() {
      const row = document.getElementById("awardsRow");
      awards.forEach(item => {
        const container = document.createElement("div");
        container.className = "flex flex-col items-center gap-2";

        if (item.svg) {
          const wrapper = document.createElement("div");
          wrapper.className = "text-indigo-700";
          wrapper.innerHTML = item.svg;
          wrapper.setAttribute("title", item.label);
          wrapper.setAttribute("aria-label", item.label);
          container.appendChild(wrapper);
        } else {
          const img = document.createElement("img");
          img.alt = item.label;
          img.title = item.label + (item.caption ? ` — ${item.caption}` : "");
          img.loading = "lazy";
          img.referrerPolicy = "no-referrer";
          img.src = logoSrcFor(item);

          // Fallback if the logo URL 404s
          img.onerror = () => {
            const fb = document.createElement("div");
            fb.className = "fallback-badge";
            fb.textContent = item.label;
            img.replaceWith(fb);
          };

          container.appendChild(img);
        }

        if (item.caption) {
          const cap = document.createElement("div");
          cap.className = "text-xs text-slate-600 text-center";
          cap.textContent = item.caption;
          container.appendChild(cap);
        }

        row.appendChild(container);
      });
    }
    renderAwards();
  </script>
